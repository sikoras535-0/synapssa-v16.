<script>
    const SARA_MAIL = "sikorasara135@gmail.com";

    function initSystem() {
        const id = document.getElementById('user-name').value;
        const role = document.getElementById('user-role').value;
        if(!id) return;

        document.getElementById('display-id').innerText = id;
        document.getElementById('display-role').innerText = role;
        document.getElementById('gate-screen').style.display = 'none';
        document.getElementById('app').style.display = 'block';
        document.getElementById('bg-narcyz').style.opacity = '0';
        document.getElementById('bg-dog').style.opacity = '1';

        loadProgressPanel(role);
    }

    function loadProgressPanel(role) {
        const area = document.getElementById('progress-area');
        let content = "";

        // Definicja zadań dla ról
        const tasks = {
            "BENEFICJENT": ["Statut Fundacji", "Deklaracja Etyki"],
            "UPRAWNIONA": ["Analiza AI Act", "Protokół Uprawnionej", "Certyfikat Dostępu"],
            "ADMIN": ["Zarządzanie Systemem", "Weryfikacja Beneficjentów", "Baza Danych R&D"]
        };

        const currentTasks = tasks[role] || tasks["BENEFICJENT"];
        
        content += `<div style="margin-bottom:15px;">
            <div style="display:flex; justify-content:space-between; font-size:0.8rem;"><span>POSTĘP ŚCIEŻKI: ${role}</span> <span id="p-val">0%</span></div>
            <div class="progress-container"><div id="p-bar" class="progress-bar"></div></div>
            <ul class="docs-list">`;
        
        currentTasks.forEach((t, index) => {
            let percent = Math.round(((index + 1) / currentTasks.length) * 100);
            content += `<li class="doc-item" onclick="setProg(${percent}, this)">${index+1}. ${t} <span>[ZATWIERDŹ]</span></li>`;
        });

        content += `</ul></div>`;
        area.innerHTML = content;
    }

    function setProg(v, el) {
        if(el.classList.contains('done')) return;
        document.getElementById('p-bar').style.width = v + '%';
        document.getElementById('p-val').innerText = v + '%';
        el.classList.add('done');
        el.innerHTML += " ✓";
    }

    function startBreathing() {
        window.speechSynthesis.cancel();
        const m = new SpeechSynthesisUtterance("Wdech. Wstrzymaj. Wydech.");
        m.lang = 'pl-PL';
        window.speechSynthesis.speak(m);
    }

    function sendMail() {
        const t = document.getElementById('rep-box').value;
        window.location.href = `mailto:${SARA_MAIL}?subject=Raport&body=${encodeURIComponent(t)}`;
    }
