<script>
    // WEJŚCIE DO SYSTEMU
    function unlock() {
        if(document.getElementById('pass').value === 'Synapssa2026') {
            document.getElementById('gate').style.display = 'none';
            document.getElementById('app').style.display = 'block';
            speak("Witaj w systemie suwerenności emocjonalnej. Jestem gotowy do Twojej dyspozycji.");
        } else {
            alert("KLUCZ NIEPRAWIDŁOWY. DOSTĘP ZABLOKOWANY.");
        }
    }

    // ZMIANA WIADOMOŚCI I GŁOS AI
    function changeMessage(txt) {
        document.getElementById('message').innerText = '"' + txt + '"';
        speak(txt);
    }

    function speak(customText) {
        // Zatrzymujemy inne mowy, jeśli są aktywne
        window.speechSynthesis.cancel(); 

        const text = customText || document.getElementById('message').innerText.replace(/"/g, "");
        const msg = new SpeechSynthesisUtterance();
        msg.text = text;
        msg.lang = 'pl-PL';
        msg.pitch = 0.9; // Spokojny, niższy ton
        msg.rate = 0.85; // Wolniejsze, kojące tempo
        window.speechSynthesis.speak(msg);
    }
</script>
