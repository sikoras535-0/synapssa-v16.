    function enter() {
        if(document.getElementById('pass').value === 'Synapssa2026') {
            document.getElementById('gate').style.display = 'none';
            document.getElementById('app').style.display = 'block';
        } else { alert('Błędny klucz'); }

    function userLogin() {
        const u = document.getElementById('user-login').value;
        if(u) alert('Witaj ' + u + '! Twoja sesja testowa została uruchomiona.');
        else alert('Podaj login');
    }

    function toggleMusic() {

        const b = document.getElementById('musicBtn');

        else { m.pause(); b.innerText = Graj Harfę; b.style.background = #4a90e2; }
