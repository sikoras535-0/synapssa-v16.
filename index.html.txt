<script>
    let currentRole = "";

    function login(role) {
        const p = document.getElementById('master-pass').value;
        if(p === 'Synapssa2026') {
            currentRole = role;
            document.getElementById('gate').style.display = 'none';
            document.getElementById('app').style.display = 'block';
            document.getElementById('role-display').innerText = role;
            
            if(role === 'ADMIN') document.getElementById('admin-only').style.display = 'block';
            
            agentSpeak("Autoryzacja pomyślna. Witaj w bezpiecznej strefie Synapssa. Jestem Twoim wsparciem.");
        } else { alert("Błędne hasło"); }
    }

    function agentSpeak(customTxt) {
        window.speechSynthesis.cancel();
        const txt = customTxt || document.getElementById('agent-text').innerText;
        const msg = new SpeechSynthesisUtterance(txt);
        msg.lang = 'pl-PL';
        msg.pitch = 0.9;
        msg.rate = 2.0; // SZYBKIE TEMPO GŁOSU
        window.speechSynthesis.speak(msg);
    }

    function toggleHarp() {
        const h = document.getElementById('harp');
        if(h.paused) h.play(); else h.pause();
    }
</script>
