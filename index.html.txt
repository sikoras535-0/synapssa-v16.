    function setupPath(role) {

                <button class="btn-gold-italic" style="font-size: 0.7rem; padding: 5px;">Zapisz w Bazie</button>
            `;
            agentSpeak("Witaj Szefowo. System gotowy do zarządzania strategicznego.");
        } 
        else if(role === 'USER_') {
            txt = `
                <p><b>Status: Ścieżka Mindfulness</b></p>
                <p style="font-size: 0.85rem;">Twój plan na dziś: 15 min medytacji Theta oraz moduł "Oddech Suwerenny".</p>
                <button class="btn-gold-italic" style="font-size: 0.7rem; padding: 5px;">Otwórz Moduł Edukacyjny</button>
            `;
            agentSpeak("Witaj. Skup się na oddechu. Twoja ścieżka rozwoju jest aktywna.");
        } 
        else if(role === 'UPRAWNIONA') {
            txt = `
                <p><b>Status: Ścieżka Premium</b></p>
                <p style="font-size: 0.85rem;">Dostęp do certyfikowanych szkoleń IT i partnerstwa biznesowego.</p>

            `;
            agentSpeak("Dzień dobry. Moduł uprawnień premium został odblokowany.");
        }
        content.innerHTML = txt;
    }

    function agentSpeak(msg) {
        window.speechSynthesis.cancel();
        const utterance = new SpeechSynthesisUtterance(msg);
        utterance.lang = 'pl-PL';
        utterance.rate = 2.0; // SZYBKI GŁOS
        utterance.pitch = 0.9;
        window.speechSynthesis.speak(utterance);
    }

    function toggleHarp() {
        const h = document.getElementById('harp');
        if(h.paused) h.play(); else h.pause();
    }
