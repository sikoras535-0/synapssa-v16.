<script>
    const SARA_MAIL = "sikorasara135@gmail.com";
    let db = JSON.parse(localStorage.getItem('syn_db_v31')) || {
        "Admin_Sara": { role: "ADMIN", pass: "Synapssa2026", progress: 0, email: SARA_MAIL }
    };

    function toggle(s) {
        document.getElementById('reg-form').style.display = s ? 'block' : 'none';
        document.getElementById('log-form').style.display = s ? 'none' : 'block';
    }

    function register() {
        const id = document.getElementById('r-id').value;
        const em = document.getElementById('r-mail').value;
        const ps = document.getElementById('r-pass').value;
        const ro = document.getElementById('r-role').value;
        if(!id || !em || !ps) return;
        db[id] = { role: ro, pass: ps, progress: 0, email: em };
        localStorage.setItem('syn_db_v31', JSON.stringify(db));
        window.location.href = `mailto:${SARA_MAIL}?cc=${em}&subject=REJESTRACJA: ${id}&body=DANE: ${id} / ${ps} / ${ro}`;
        alert("Zarejestrowano. Sprawdź maila i zaloguj się.");
        toggle(false);
    }

    function login() {
        const id = document.getElementById('l-id').value;
        const ps = document.getElementById('l-pass').value;
        if(db[id] && db[id].pass === ps) {
            sessionStorage.setItem('user', id);
            showApp(id);
        } else { alert("Błąd!"); }
    }

    function showApp(id) {
        document.getElementById('auth-screen').style.display = 'none';
        document.getElementById('app').style.display = 'block';
        document.getElementById('d-id').innerText = id + " [" + db[id].role + "]";
        
        const zone = document.getElementById('action-zone');
        if(db[id].role === 'ADMIN') {
            document.getElementById('admin-panel').style.display = 'block';
            renderAdmin();
            zone.innerHTML = `<input id="ad-to" placeholder="Do kogo..."><textarea id="ad-m"></textarea><button class="btn-gold" onclick="adminSend()">WYŚLIJ JAKO SARA</button>`;
        } else {
            updateBar(id);
            zone.innerHTML = `<textarea id="u-m" placeholder="Materiały do Pani Sary..."></textarea><button class="btn-gold" onclick="userSend('${id}')">WYŚLIJ MATERIAŁY</button>`;
        }
    }

    function renderAdmin() {
        const l = document.getElementById('u-list');
        l.innerHTML = Object.keys(db).map(u => `<div>ID: ${u} | Rola: ${db[u].role} | Postęp: ${db[u].progress}%</div>`).join('');
    }

    function action(m) {
        const id = sessionStorage.getItem('user');
        if(db[id].progress < 100) db[id].progress += 1;
        localStorage.setItem('syn_db_v31', JSON.stringify(db));
        updateBar(id);
        alert(m + " | +1%");
    }

    function updateBar(id) {
        document.getElementById('p-bar').style.width = db[id].progress + "%";
        document.getElementById('p-val').innerText = db[id].progress + "%";
    }

    function userSend(id) {
        const m = document.getElementById('u-m').value;
        action('Wysłano materiał');
        window.location.href = `mailto:${SARA_MAIL}?subject=Materiały: ${id}&body=${encodeURIComponent(m)}`;
    }

    function adminSend() {
        const t = document.getElementById('ad-to').value;
        const m = document.getElementById('ad-m').value;
        window.location.href = `mailto:${t}?cc=${SARA_MAIL}&subject=Admin Synapssa&body=${encodeURIComponent(m)}`;
    }

    function toggleForest() {
        const f = document.getElementById('forest');
        f.paused ? f.play() : f.pause();
    }

    function startBreathing() {
        window.speechSynthesis.cancel();
        const m = new SpeechSynthesisUtterance("Spokojnie, oddychaj na trzy. Wdech: raz, dwa, trzy. Wstrzymaj: raz, dwa, trzy. Wydech: raz, dwa, trzy.");
        m.lang = 'pl-PL'; m.rate = 1.5; m.pitch = 1.2;
        window.speechSynthesis.speak(m);
    }
</script>
