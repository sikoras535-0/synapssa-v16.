<script>
    // BAZA DANYCH (LOCALSTORAGE)
    window.onload = () => {
        const savedRole = localStorage.getItem('synapssa_role');
        if(savedRole) {
            currentRole = savedRole;
            showApp(savedRole);
        }
    };

    function login(role) {
        if(document.getElementById('master-pass').value === 'Synapssa2026') {
            localStorage.setItem('synapssa_role', role);
            showApp(role);
        } else { alert("NIEAUTORYZOWANY DOSTĘP"); }
    }

    function logout() {
        localStorage.removeItem('synapssa_role');
        location.reload();
    }

    function showApp(role) {
        document.getElementById('gate').style.display = 'none';
        document.getElementById('app').style.display = 'block';
        document.getElementById('role-display').innerText = role;
        
        const content = document.getElementById('path-content');
        if(role === 'ADMIN') {
            content.innerHTML = `<p><b>ZARZĄDZANIE:</b> Masz pełne uprawnienia do wgrywania plików i modyfikacji bazy.</p><input type="file"><button class="btn-gold-italic" style="font-size:0.7rem">WGRAJ DO CHMURY</button>`;
        } else if(role === 'USER_') {
            content.innerHTML = `<p><b>TERAPIA:</b> Ścieżka mindfulness aktywna. Ukończ dzisiejszą sesję lasu.</p><button class="btn-gold-italic" style="font-size:0.7rem">DZIENNIK SPOKOJU</button>`;
        } else {
            content.innerHTML = `<p><b>PREMIUM:</b> Dostęp do certyfikatów IT i partnerstwa strategicznego.</p><button class="btn-gold-italic" style="font-size:0.7rem">POBIERZ CERTYFIKAT</button>`;
        }
    }

    // INTERAKTYWNY GŁOS (KOBIETA, TEMPO 1.5)
    function startBreathing() {
        window.speechSynthesis.cancel();
        const text = "Spokojnie. Oddychaj na trzy. Wdech: raz, dwa, trzy. Wstrzymaj: raz, dwa, trzy. Wydech: raz, dwa, trzy.";
        const msg = new SpeechSynthesisUtterance(text);
        
        // Próba wybrania głosu żeńskiego (zależy od przeglądarki)
        const voices = window.speechSynthesis.getVoices();
        const femaleVoice = voices.find(v => v.name.includes('Google') && v.lang.includes('pl')) || voices.find(v => v.lang.includes('pl'));
        
        if(femaleVoice) msg.voice = femaleVoice;
        msg.lang = 'pl-PL';
        msg.rate = 1.5; // TEMPO 1.5
        msg.pitch = 1.1; // Nieco wyższy głos kobiecy
        
        window.speechSynthesis.speak(msg);
    }

    function toggleForest() {
        const audio = document.getElementById('forest-sound');
        if(audio.paused) audio.play(); else audio.pause();
    }
</script>
