<script>
    // Obiekt przechowujący postęp
    let progress = { s1: 0, s2: 0, s3: 0 };

    function completeStep(sciezka, value, element) {
        if(element.classList.contains('done')) return; // Zapobieganie cofaniu

        // Ustawianie postępu
        if(value > progress[sciezka]) {
            progress[sciezka] = value;
            updateUI(sciezka);
        }

        // Oznaczanie dokumentu jako przeczytany
        element.classList.add('done');
        element.querySelector('.status-tag').innerText = "ZATWIERDZONO";

        // Symulacja otwarcia dokumentu
        console.log("Otwarto dokument w " + sciezka + ". Postęp: " + value + "%");
    }

    function updateUI(s) {
        const bar = document.getElementById(s + '-bar');
        const valText = document.getElementById(s + '-val');
        
        bar.style.width = progress[s] + '%';
        valText.innerText = progress[s] + '%';

        if(progress[s] === 100) {
            valText.innerHTML = " ukończona ✓";
            valText.style.color = "#2ecc71";
        }
    }
</script>
