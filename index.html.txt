<script>
    let BAZA_HASEL = [];

    function login() {
        const id = document.getElementById('reg-id').value;
        const pass = document.getElementById('reg-pass').value;
        const role = document.getElementById('reg-role').value;
        if(!id || !pass) return;

        // Dodanie do bazy haseł
        BAZA_HASEL.push({user: id, pass: pass, role: role, time: new Date().toLocaleTimeString()});
        updatePassDB();

        document.getElementById('u-name').innerText = id;
        document.getElementById('u-role').innerText = role;
        document.getElementById('gate-screen').style.display = 'none';
        document.getElementById('app').style.display = 'block';
        document.getElementById('bg-narcyz').style.opacity = '0';
        document.getElementById('bg-dog').style.opacity = '1';

        if(role === "ADMIN") document.getElementById('admin-panel').style.display = 'block';
    }

    function updatePassDB() {
        const dbDiv = document.getElementById('pass-db');
        dbDiv.innerHTML = BAZA_HASEL.map(entry => `[${entry.time}] ${entry.role}: ${entry.user} | PWD: ${entry.pass}`).join('<br>');
    }

    function sendCert() {
        const id = document.getElementById('reg-id').value;
        window.location.href = `mailto:sikorasara135@gmail.com?subject=CERTYFIKAT&body=Uzytkownik ${id} prosi o haslo.`;
    }

    function sendMail() {
        const txt = document.getElementById('mail-body').value;
        window.location.href = `mailto:sikorasara135@gmail.com?subject=RAPORT&body=${encodeURIComponent(txt)}`;
    }

    function adminToUser() {
        const mail = document.getElementById('target-mail').value;
        const txt = document.getElementById('admin-text').value;
        window.location.href = `mailto:${mail}?subject=WIADOMOSC OD ADMINA&body=${encodeURIComponent(txt)}`;
    }

    function updateP(id, val) {
        document.getElementById(id).style.width = val + '%';
    }

    function startBreathing() {
        window.speechSynthesis.cancel();
        const m = new SpeechSynthesisUtterance("Spokojnie, oddychaj na trzy. Wdech: raz, dwa, trzy. Wstrzymaj: raz, dwa, trzy. Wydech: raz, dwa, trzy. Powtarzaj w spokoju.");
        m.lang = 'pl-PL';
        window.speechSynthesis.speak(m);
    }
</script>
