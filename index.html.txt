<script>
    // BAZA DANYCH: Nienaruszalna tablica przetrzymywana w przeglądarce (skala masowa)
    let DB = JSON.parse(localStorage.getItem('synapssa_ultimate_db')) || [];

    // GŁÓWNA FUNKCJA LOGOWANIA (PŁYNNE PRZEJŚCIE)
    function login() {
        const id = document.getElementById('reg-id').value;
        const pass = document.getElementById('reg-pass').value;
        const role = document.getElementById('reg-role').value;
        
        if(!id || !pass) {
            alert("Suwerenie, podaj ID i Hasło.");
            return;
        }

        // Zapis do bazy danych haseł i progresu
        let user = DB.find(u => u.id === id);
        if(!user) {
            user = {id: id, pass: pass, role: role, pDuch: 0, pWiedza: 0, pCzyn: 0};
            DB.push(user);
            saveDB();
        } else {
            // Aktualizacja hasła jeśli wpisał inne dla tego samego loginu (logika bazy haseł)
            user.pass = pass; 
            user.role = role;
            saveDB();
        }

        // Ustawienie UI dla zalogowanego
        document.getElementById('u-name').innerText = id;
        document.getElementById('u-role').innerText = role;
        
        // Płynne przenikanie tła i ekranów
        document.getElementById('gate-screen').style.opacity = '0';
        setTimeout(() => {
            document.getElementById('gate-screen').style.display = 'none';
            document.getElementById('app').style.display = 'block';
            document.getElementById('bg-narcyz').style.opacity = '0';
            document.getElementById('bg-dog').style.opacity = '1';
        }, 800); // Czas trwania animacji krycia

        // Załadowanie indywidualnych pasków postępu usera
        setTimeout(() => {
            document.getElementById('p-duch').style.width = user.pDuch + '%';
            document.getElementById('txt-duch').innerText = user.pDuch;
            document.getElementById('p-wiedza').style.width = user.pWiedza + '%';
            document.getElementById('txt-wiedza').innerText = user.pWiedza;
            document.getElementById('p-czyn').style.width = user.pCzyn + '%';
        }, 1000); // Małe opóźnienie dla ładnego efektu nalewania

        // Ujawnienie panelu Admina
        if(role === "ADMIN") {
            document.getElementById('admin-panel').style.display = 'block';
            refreshAdminTable();
        }
    }

    function saveDB() { 
        localStorage.setItem('synapssa_ultimate_db', JSON.stringify(DB)); 
    }

    // PŁYNNE AKTUALIZOWANIE POSTĘPU
    function updateUserProgress(barId, val, element = null) {
        // UI usera
        document.getElementById(barId).style.width = val + '%';
        
        // Aktualizacja procentów tekstowych
        if(barId === 'p-duch') document.getElementById('txt-duch').innerText = val;
        if(barId === 'p-wiedza') document.getElementById('txt-wiedza').innerText = val;

        // Oznaczenie dokumentu w liście pozycjonowanej
        if(element) {
            element.classList.add('done');
            element.querySelector('span').innerText = "[ZATWIERDZONO ✓]";
        }

        // Zapis do bazy
        const name = document.getElementById('u-name').innerText;
        let user = DB.find(u => u.id === name);
        if(user) {
            if(barId === 'p-duch') user.pDuch = val;
            if(barId === 'p-wiedza') user.pWiedza = val;
            if(barId === 'p-czyn') user.pCzyn = val;
            saveDB();
        }

        // Odświeżenie tabeli admina w tle, jeśli admin jest zalogowany
        if(document.getElementById('u-role').innerText === "ADMIN") {
            refreshAdminTable();
        }
    }

    // RENDEROWANIE TABELI DLA 50 000 UŻYTKOWNIKÓW Z BAZĄ HASEŁ I PASKAMI
    function refreshAdminTable() {
        const tbody = document.getElementById('admin-table-body');
        tbody.innerHTML = DB.map(u => `
            <tr>
                <td><b>${u.id}</b></td>
                <td style="color:#888; font-size:0.7rem;">${u.role}</td>
                <td><code style="color:var(--gold); letter-spacing:1px; background:#111; padding:3px 6px; border-radius:3px;">${u.pass}</code></td>
                <td>
                    <div style="display:flex; align-items:center; gap:10px;">
                        <div class="progress-container" style="margin:0; height:10px;"><div class="progress-bar" style="width:${u.pDuch}%"></div></div>
                        <span style="font-size:0.7rem; color:var(--gold); width:30px;">${u.pDuch}%</span>
                    </div>
                </td>
                <td>
                    <div style="display:flex; align-items:center; gap:10px;">
                        <div class="progress-container" style="margin:0; height:10px;"><div class="progress-bar" style="width:${u.pWiedza}%"></div></div>
                        <span style="font-size:0.7rem; color:var(--gold); width:30px;">${u.pWiedza}%</span>
                    </div>
                </td>
            </tr>
        `).join('');
    }

    // NIE SKRÓCONY TRENING ODDECHOWY
    function startBreathing() {
        window.speechSynthesis.cancel();
        const m = new SpeechSynthesisUtterance("Spokojnie, oddychaj na trzy. Wdech: raz, dwa, trzy. Wstrzymaj: raz, dwa, trzy. Wydech: raz, dwa, trzy. Powtarzaj w spokoju suwerena.");
        m.lang = 'pl-PL'; 
        m.rate = 0.85; // Wolniejsze, płynne tempo
        m.pitch = 0.9;
        window.speechSynthesis.speak(m);
    }
